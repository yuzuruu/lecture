summarize to obtain aggregated data

data_seniority_summary <- 
  data_seniority %>% 
  dplyr::filter(
    school == "university"
  ) %>% 
  group_by(length_service, age_class) %>% 
  summarise(
    Mean = mean(value, na.rm = TRUE)
  )
